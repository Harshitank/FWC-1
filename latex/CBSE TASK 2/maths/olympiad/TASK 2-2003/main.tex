\begin{enumerate}

\item $S$ is the set $\{1,2,3,\dots,1000000\}$.  
Show that for any subset $A$ of $S$ with $101$ elements we can find $100$ distinct elements $x_i$ of $S$ such that the sets $\{a+x_i \mid a \in A\}$ are all pairwise disjoint.

\item Find all pairs $(m,n)$ of positive integers such that
\[
\frac{m^2}{2m^2 - n^2 + 1}
\]
is a positive integer.

\item A convex hexagon has the property that for any pair of opposite sides the distance between their midpoints is $\sqrt{3}/2$ times the sum of their lengths.
Show that all the hexagonâ€™s angles are equal.

\item $ABCD$ is cyclic.  
The feet of the perpendiculars from $D$ to the lines $AB, BC, CA$ are $P,Q,R$ respectively.
Show that the angle bisectors of $\angle ABC$ and $\angle CDA$ meet on the line $AC$ if and only if
\[
RP = RQ.
\]

\item Given $n>2$ and reals $x_1 \le x_2 \le \dots \le x_n$, show that
\[
\sum_{i<j} |x_i - x_j|^2 \le \frac{n^2-1}{3}\sum_{i<j}(x_i-x_j)^2.
\]
Show that equality holds if and only if the sequence is an arithmetic progression.

\item Show that for each prime $p$, there exists a prime $q$ such that
\[
n^p - p
\]
is not divisible by $q$ for any positive integer $n$.

\end{enumerate}
