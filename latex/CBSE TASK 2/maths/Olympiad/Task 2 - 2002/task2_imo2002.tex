%================ QUESTIONS =================%
\begin{enumerate}[leftmargin=0pt, label=\arabic*., itemsep=12pt]

\item Let $S$ be the set of all $(h,k)$ with $h,k$ non--negative integers such that
\begin{equation*}
\begin{aligned}[t]
h + k < n .
\end{aligned}
\end{equation*}

Each element of $S$ is colored red or blue, so that if $(h,k)$ is red and
\begin{equation*}
\begin{aligned}[t]
h' \le h,\quad k' \le k,
\end{aligned}
\end{equation*}
then $(h',k')$ is also red.

A type--1 subset of $S$ has $n$ blue elements with different first members, and
a type--2 subset of $S$ has $n$ blue elements with different second members.

Show that there are the same number of type--1 and type--2 subsets.

\item $BC$ is a diameter of a circle with center $O$.
$A$ is any point on the circle such that
\begin{equation*}
\begin{aligned}[t]
\angle AOC > 60^\circ .
\end{aligned}
\end{equation*}

$EF$ is the chord which is the perpendicular bisector of $AO$.
$D$ is the midpoint of the minor arc $AB$.
The line through $O$ parallel to $AD$ meets $AC$ at $J$.

Show that $J$ is the incenter of triangle $CEF$.

\item Find all pairs of integers $m > 2$ and $n > 2$ such that there are infinitely
many positive integers $k$ for which
\begin{equation*}
\begin{aligned}[t]
k^n + k^2 - 1 \mid k^m + k - 1 .
\end{aligned}
\end{equation*}

\item Let the positive divisors of an integer $n > 1$ be
$d_1 < d_2 < \cdots < d_k$, where $d_1 = 1$ and $d_k = n$.
Let
\begin{equation*}
\begin{aligned}[t]
d = d_1 d_2 + d_2 d_3 + \cdots + d_{k-1} d_k .
\end{aligned}
\end{equation*}

Show that $d < n^2$ and determine all $n$ for which $d$ divides $n^2$.

\item Find all real--valued functions $f$ defined on $\mathbb{R}$ such that
\begin{equation*}
\begin{aligned}[t]
f(x) + f(y)\bigl(f(f(x)) + f(y)\bigr)
= f(x - y) + f(xy + y)
\end{aligned}
\end{equation*}
for all real $x,y$.

\item Let $a,b,c,d$ be integers such that
\begin{equation*}
\begin{aligned}[t]
a > b > c > d > 0
\end{aligned}
\end{equation*}
and
\begin{equation*}
\begin{aligned}[t]
ac + bd = (b + d + a - c)(b + d - a + c).
\end{aligned}
\end{equation*}

Prove that $a + b + c + d$ is not prime.

\end{enumerate}
