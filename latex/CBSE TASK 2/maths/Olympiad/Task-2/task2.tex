%================ QUESTIONS =================%
\begin{enumerate}[leftmargin=0pt, label=\arabic*., itemsep=12pt]

\item Let $ABC$ be an acute-angled triangle with circumcentre $O$.
Let $P$ be the foot of the altitude from $A$ on $BC$.

Suppose that
\begin{equation*}
\begin{aligned}[t]
\angle BCA \ge \angle ABC + 30^\circ .
\end{aligned}
\end{equation*}

Prove that
\begin{equation*}
\begin{aligned}[t]
\angle CAB + \angle COP < 90^\circ .
\end{aligned}
\end{equation*}

\item Prove that
\begin{equation*}
\begin{aligned}[t]
\frac{a}{\sqrt{a^2+8bc}} +
\frac{b}{\sqrt{b^2+8ca}} +
\frac{c}{\sqrt{c^2+8ab}} \ge 1
\end{aligned}
\end{equation*}
for all positive real numbers $a,b,c$.

\item Twenty-one girls and twenty-one boys took part in a mathematical contest.
\begin{itemize}
\item Each contestant solved at most six problems.
\item For each girl and each boy, at least one problem was solved by both of them.
\end{itemize}

Prove that there was a problem that was solved by at least three girls and at least three boys.

\item Let $n$ be an odd integer greater than $1$ and
$k_1, k_2, \dots, k_n$ be integers.

For each permutation
\begin{equation*}
\begin{aligned}[t]
a = (a_1, a_2, \dots, a_n)
\end{aligned}
\end{equation*}
of $1,2,\dots,n$, define
\begin{equation*}
\begin{aligned}[t]
S(a) = \sum_{i=1}^{n} k_i a_i .
\end{aligned}
\end{equation*}

Prove that there exist two distinct permutations $b$ and $c$ such that
\begin{equation*}
\begin{aligned}[t]
n! \mid (S(b) - S(c)).
\end{aligned}
\end{equation*}

\item In triangle $ABC$, let $AP$ bisect $\angle BAC$ and
$BQ$ bisect $\angle ABC$.

Given that
\begin{equation*}
\begin{aligned}[t]
\angle BAC = 60^\circ
\quad \text{and} \quad
AB + BP = AQ + QB,
\end{aligned}
\end{equation*}
find the possible angles of triangle $ABC$.

\item Let $a,b,c,d$ be integers such that
\begin{equation*}
\begin{aligned}[t]
a > b > c > d > 0
\end{aligned}
\end{equation*}
and
\begin{equation*}
\begin{aligned}[t]
ac + bd = (b + d + a - c)(b + d - a + c).
\end{aligned}
\end{equation*}

Prove that $a + b + c + d$ is not prime.

\end{enumerate}
